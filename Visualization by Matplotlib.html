2. Bar Chart¶

In [70]:
# Create list of 3 variables to analyze the companies revenue and profit
Company=['Zoho','Tiger Analytics','Freshworks','Mphasis']
Revenue=[1000,1250,1240,1850]
Profit=[100,200,150,300]
In [71]:
import numpy as np
xpos=np.arange(len(Company)) # create a numpy array to convert strings to numbers
plt.xticks(xpos,Company) # create labels for x-axis
plt.ylabel("Revenue",fontsize=12)
plt.title("Tech Revenues",fontsize=15)
plt.bar(xpos-0.2,Revenue,width=0.4,label="Revenue") # xpos-0.2 move the profit bar to a side
plt.bar(xpos+0.2,Profit,width=0.4,label="Profit")
plt.legend(loc="best")
